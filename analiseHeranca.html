Código do Aluno

<?php
class Pessoa {
    protected $nome;
    protected $idade;
    public function __construct($nome, $idade) {
        $this->nome = $nome;
        $this->idade = $idade;
    }
}

class Aluno extends Pessoa {
    private $matricula;

    public function __construct($nome, $idade, $matricula) {
        parent::__construct($nome, $idade);
        $this->matricula = $matricula;
    }

    public function apresentar() {
        echo "O aluno(a) {$this->nome} tem {$this->idade} anos e a matrícula {$this->matricula}.";
    }
}
class Professor extends Pessoa {
    private $materia;

    public function __construct($nome, $idade, $materia) {
        parent::__construct($nome, $idade);
        $this->materia = $materia;
    }

    public function apresentar() {
        echo "O professor {$this->nome}, de {$this->idade} anos, da a aula de {$this->materia}.";
    }
}

$aluno = new Aluno("Mykaella", 18, "A01234");
$professor = new Professor("Gabriel", 25, "Educação Física");

$aluno->apresentar();
echo "<br>";
$professor->apresentar();
?>
